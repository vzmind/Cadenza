def eval(context)
  l = self.left.eval(context)
  r = self.right.eval(context)
  
  case self.op
    when '=='
      return l == r
      
    when '!='
      return l != r
      
    when '>='
      return l >= r
      
    when '<='
      return l <= r
      
    when '>'
      return l > r
    
    when '<'
      return l < r
  end
end
